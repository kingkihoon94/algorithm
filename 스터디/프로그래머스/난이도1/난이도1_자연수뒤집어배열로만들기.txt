class Solution {
    public int[] solution(long n) {
        String tmp = "" + n;
        int[] answer = new int[tmp.length()];
        int cnt=0;
        while(n>0) {
            answer[cnt]=(int)(n%10);
            n/=10;
            cnt++;
        }//end while.
        return answer;
    }//end main.
}//end solution.
